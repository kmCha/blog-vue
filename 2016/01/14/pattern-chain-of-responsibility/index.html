javascript设计模式之职责链模式(chain of responsibility)

发表于 2016-01-14   |   分类于 javascript   |  

职责链是什么？

如果我们有一个请求需要其他对象处理，那么将可能处理这个请求的对象连成一根链条，将请求在这根链条上向下传递，直到有一个对象处理这个请求为止，这种设计模式就叫做职责链模式。要理解这个模式的好处，就先来看看例子。

一道编程题(FizzBuzzWhizz)

下面这个编程题来自之前拉勾网上的一个面试活动，这里仅仅为了展示职责链，代码可能写得有点丑。

FizzBuzzWhizz
你是一名体育老师，在某次课距离下课还有五分钟时，你决定搞一个游戏。此时有100名学生在上课。游戏的规则是：

  1. 你首先说出三个不同的特殊数，要求必须是个位数，比如3、5、7。
  2. 让所有学生拍成一队，然后按顺序报数。
  3. 学生报数时，如果所报数字是第一个特殊数（3）的倍数，那么不能说该数字，而要说Fizz；如果所报数字是第二个特殊数（5）的倍数，那么要说Buzz；如果所报数字是第三个特殊数（7）的倍数，那么要说Whizz。
  4. 学生报数时，如果所报数字同时是两个特殊数的倍数情况下，也要特殊处理，比如第一个特殊数和第二个特殊数的倍数，那么不能说该数字，而是要说FizzBuzz, 以此类推。如果同时是三个特殊数的倍数，那么要说FizzBuzzWhizz。
  5. 学生报数时，如果所报数字包含了第一个特殊数，那么也不能说该数字，而是要说相应的单词，比如本例中第一个特殊数是3，那么要报13的同学应该说Fizz。如果数字中包含了第一个特殊数，那么忽略规则3和规则4，比如要报35的同学只报Fizz，不报BuzzWhizz。
    现在，我们需要你完成一个程序来模拟这个游戏，它首先接受3个特殊数，然后输出100名学生应该报数的数或单词。比如，
    输入

    Buzz
    Fizz
    Whizz
    8

    …
    一直到100

这里重要的是不要被那句有些人10行就可以搞定迷惑，你不信你交个10行的代码给他试试他让不让你通过。这里的重点当然是考察面相对象和函数式编程的功底，编写出易读、易维护、可重用的代码才是应该考虑的，单元测试应该也是重点，这里主要想说的是职责链，因此没有编写单元测试代码（其实是懒）。

炮灰写法

那么先来看第一段代码，编写一个全局函数传入全班同学的个数（这里是100）和三个特殊数字以及对应的输出文本，这个函数会返回一个有go()方法的对象，go()方法会返回最后输出的文本的数组。

// fbw.html
<script type="text/javascript">
    window.onload = function() {
        var fbw = createFBW(100, [
                {number: 3, string: "Fizz"},
                {number: 5, string: "Buzz"},
                {number: 7, string: "Whizz"}
            ]),
            list = fbw.go();
        console.log(list);
    };
</script>

// fbw.js
var createFBW = function(max, input) {
    var inputNum = [input[0].number, input[1].number, input[2].number],
        list = [],
        str1 = input[0].string || "Fizz",
        str2 = input[1].string || "Buzz",
        str3 = input[2].string || "Whizz",
        go = function() {
            for(var i = 1; i <= max; i++) {
                if(i.toString().indexOf(inputNum[0]) >= 0) {
                    list.push(str1);
                }
                else if((i % inputNum[0] === 0) && (i % inputNum[1] === 0) && (i % inputNum[2] === 0)) {
                    list.push(str1 + str2 + str3);
                }
                else if((i % inputNum[0] === 0) && (i % inputNum[1] === 0)) {
                    list.push(str1 + str2);
                }
                else if((i % inputNum[0] === 0) && (i % inputNum[2] === 0)) {
                    list.push(str1 + str3);
                }
                else if((i % inputNum[1] === 0) && (i % inputNum[2] === 0)) {
                    list.push(str2 + str3);
                }
                else if(i % inputNum[0] === 0) {
                    list.push(str1);
                }
                else if(i % inputNum[1] === 0) {
                    list.push(str2);
                }
                else if(i % inputNum[2] === 0) {
                    list.push(str3);
                }
                else {
                    list.push(i);
                }
            }
            return list;
        };

    return {
        go: go
    };
};

执行程序之后，打开控制台就能看见输出的数组，说明代码确实能够正常工作。那么现在把目光转移到代码身上，来看看这样写问题在哪。其实不用说也很明显了，当然是含有巨量if分支的go()函数了，首先有这么多的条件判断会让代码难以阅读，其次go()函数就是个输出数组的函数，让他做这么多的条件判断不太合适，不符合单一职责原则。因此我们的思路应该是将几个不同的规则封装成函数，在函数内部进行逻辑判断。

OOP职责链

下面用面向对象的职责链模式来重构代码：

// fbw.js
var createFBW = (function() {
    var containsFirst = function(num, inputNum, list, str1, str2, str3) {
        if(num.toString().indexOf(inputNum[0]) >= 0) {
            return list.push(str1);
        }
        return "next";
    };
    var multipleOfThree = function(num, inputNum, list, str1, str2, str3) {
        if((num % inputNum[0] === 0) && (num % inputNum[1] === 0) && (num % inputNum[2] === 0)) {
            return list.push(str1 + str2 + str3);
        }
        return "next";
    };
    var multipleOfTwo = function(num, inputNum, list, str1, str2, str3) {
        if((num % inputNum[0] === 0) && (num % inputNum[1] === 0)) {
            return list.push(str1 + str2);
        }
        else if((num % inputNum[0] === 0) && (num % inputNum[2] === 0)) {
            return list.push(str1 + str3);
        }
        else if((num % inputNum[1] === 0) && (num % inputNum[2] === 0)) {
            return list.push(str2 + str3);
        }
        return "next";
    };
    var multipleOfOne = function(num, inputNum, list, str1, str2, str3) {
        if((num % inputNum[0] === 0)) {
            return list.push(str1);
        }
        else if((num % inputNum[1] === 0)) {
            return list.push(str2);
        }
        else if((num % inputNum[2] === 0)) {
            return list.push(str3);
        }
        return "next";
    };
    var noneOfAbove = function(num, inputNum, list, str1, str2, str3) {
        return list.push(num);
    };

    function Chain(fn) {
        this.next = null;
        this.fn = fn;
    }
    Chain.prototype.go = function() {
        var result = this.fn.apply(this, arguments);
        if(result === "next") {
            this.next.go.apply(this.next, arguments);
        }
    };
    var rule1 = new Chain(containsFirst),
        rule2 = new Chain(multipleOfThree),
        rule3 = new Chain(multipleOfTwo),
        rule4 = new Chain(multipleOfOne),
        rule5 = new Chain(noneOfAbove);

    // 设定职责链顺序
    rule1.next = rule2;
    rule2.next = rule3;
    rule3.next = rule4;
    rule4.next = rule5;

    var createFBW = function(max, input) {
        var inputNum = [input[0].number, input[1].number, input[2].number],
            list = [],
            str1 = input[0].string || "Fizz",
            str2 = input[1].string || "Buzz",
            str3 = input[2].string || "Whizz",
            go = function() {
                for(var i = 1; i <= max; i++) {
                    rule1.go(i, inputNum, list, str1, str2, str3);
                }
                return list;
            };

        return {
            go: go
        };
    };

    return createFBW;
})();

可以看到我们将各个不同的规则封装成了几个函数。然后我们定义了一个Chain()构造函数，用来创建职责链对象，每个职责链对象中存在next和fn属性。next属性用来指定该职责链对象的下一个指向的职责链对象，这样我们可以消除几个规则函数之间的耦合，不用在函数内部指定下一个需要调用的函数。通过next属性我们也能够轻易地添加和修改职责链的结构。fn属性中装的是该职责链对象需要调用的函数，如果该函数返回某种约定的值（这里指定返回字符串“next”），那么说明该职责对象没有处理请求，将交给下一个职责链对象，以此类推直到一个函数不返回约定值时停止。
这里职责链的顺序是根据规则的优先级来确定的，根据题目描述，当数中含有第一个特殊数字的时候忽略其他规则显示相应的字符串，因此我们把containsFirst()函数对应的职责链对象rule1放在职责链的第一个，同理可得后面的顺序。这样就能保证函数执行的顺序了，通过执行rule1.go()之后，如果不满足第一个规则，则会传递给下一个职责链对象，直到有一个职责链对象的fn不返回“next”为止，这样就达到了跟第一个例子中的if分句一样的效果。

AOP职责链

还是那句话javascript中函数是一等对象，可以用对象来处理的问题我们也应该考虑函数式编程的实现，这样才对得起javascript的灵活性，下面来看下用AOP来重构上面的代码：

var createFBW = (function() {
    var containsFirst = function(num, inputNum, list, str1, str2, str3) {
        if(num.toString().indexOf(inputNum[0]) >= 0) {
            return list.push(str1);
        }
        return "next";
    };
    var multipleOfThree = function(num, inputNum, list, str1, str2, str3) {
        if((num % inputNum[0] === 0) && (num % inputNum[1] === 0) && (num % inputNum[2] === 0)) {
            return list.push(str1 + str2 + str3);
        }
        return "next";
    };
    var multipleOfTwo = function(num, inputNum, list, str1, str2, str3) {
        if((num % inputNum[0] === 0) && (num % inputNum[1] === 0)) {
            return list.push(str1 + str2);
        }
        else if((num % inputNum[0] === 0) && (num % inputNum[2] === 0)) {
            return list.push(str1 + str3);
        }
        else if((num % inputNum[1] === 0) && (num % inputNum[2] === 0)) {
            return list.push(str2 + str3);
        }
        return "next";
    };
    var multipleOfOne = function(num, inputNum, list, str1, str2, str3) {
        if((num % inputNum[0] === 0)) {
            return list.push(str1);
        }
        else if((num % inputNum[1] === 0)) {
            return list.push(str2);
        }
        else if((num % inputNum[2] === 0)) {
            return list.push(str3);
        }
        return "next";
    };
    var noneOfAbove = function(num, inputNum, list, str1, str2, str3) {
        return list.push(num);
    };

    Function.prototype.nextRule = function(fn) {
        var that = this;
        return function() {
            var result = that.apply(this, arguments);
            if(result === "next") {
                return fn.apply(this, arguments);
            }
            return result;
        };
    };
    var rule = containsFirst.nextRule(multipleOfThree).nextRule(multipleOfTwo).nextRule(multipleOfOne).nextRule(noneOfAbove);

    var createFBW = function(max, input) {
        var inputNum = [input[0].number, input[1].number, input[2].number],
            list = [],
            str1 = input[0].string || "Fizz",
            str2 = input[1].string || "Buzz",
            str3 = input[2].string || "Whizz",
            go = function() {
                for(var i = 1; i <= max; i++) {
                    rule(i, inputNum, list, str1, str2, str3);
                }
                return list;
            };

        return {
            go: go
        };
    };

    return createFBW;
})();

看上去那么大一截，实际上根OOP的例子相比，重点变化就是下面这一小部分：

Function.prototype.nextRule = function(fn) {
    var that = this;
    return function() {
        var result = that.apply(this, arguments);
        if(result === "next") {
            return fn.apply(this, arguments);
        }
    };
};
var rule = containsFirst.nextRule(multipleOfThree).nextRule(multipleOfTwo).nextRule(multipleOfOne).nextRule(noneOfAbove);

我们给所有函数对象添加了一个nextRule()方法，这个方法接收一个函数，返回一个函数。接收的函数会作为在本函数返回了约定值之后调用的函数，也就是OOP中职责链对象指向的下一个职责链对象中的fn属性。而返回的函数相当于OOP例子中的rule.go()方法。在这里，因为nextRule()返回的还是一个函数，因此我们可以用链式写法来指定职责链的顺序，就像上面的变量rule一样。变量rule引用的是从职责链中优先级最高的函数开始，经过一系列链式调用nextRule()方法而返回的函数。调用了rule()之后就会顺着职责链往下执行传入的函数，直到有一个函数没有返回约定值为止。
实际上这样的写法让我想起了装饰者模式中的before和after，他们确实很像，都是给函数指定了执行顺序，都是形成了一条执行链，但是，装饰者模式的意图是动态地给函数或对象添加功能，而这里的职责链模式并不是为了增加功能，而是为了在执行链的优先级前提下找到一个能够处理请求的函数或对象。也就是说装饰者模式是从装饰链的头一直执行到尾的，为的就是执行这些添加到装饰链上的“装饰”。而职责链模式是为了找到满足要求的对象或函数，职责链里处在该对象或函数后面的对象或函数就不用再执行了。

职责链模式的好处

我们可以通过职责链来代替大段大段的if-else语句，同时也能让不同的处理请求的函数之间解耦合。另外，在if-else语句中，我们是不能跳过中间任何的分句的，但是通过职责链模式，我们可以手动指定职责链的结构，可以指定职责链开始的位置等等。
