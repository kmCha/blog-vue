javascript事件管理与自定义事件解耦

发表于 2015-12-23   |   分类于 javascript   |  

管理事件处理程序

在javascript中，事件机制是很早就被提出了的。在DOM level0中，我们可以通过给元素的onXXXX特性（比如说onload、onclick、blablabla）赋值来实现事件处理。在非侵入式写法中就是在javascript中将该元素的onXXXX属性赋值为一个回调函数，当事件触发的时候该函数会被自动执行。简单易行的同时也存在很多问题，其中之一就是不能同时指定多个回调函数。好然后DOM level2来了，在非IE的浏览器中对应元素出现了一个addEventListener(type, fn, false)方法，IE中对应为attachEvent("on" + type, fn)方法。两个方法在实现绑定事件处理程序的过程上稍有不同，但是结果是一样的（严格来说也不一样，比如说IE中只支持事件冒泡，不支持捕获，同时IE中事件处理函数中的this是绑定到window对象上的），只需要在绑定的时候稍作判断就能确定在当前上下文中应该使用哪一个方法。重点是用一个元素连续调用该方法多次，就可以方便地绑定事件处理程序多次，事件触发时就能调用多个你指定的函数了。当然DOM level2中的事件模型当然也有问题（不然我写这个干啥）。在简单的应用中，绑定的事件处理函数不多，当你需要解除绑定的时候，你可以一个一个的用removeEventListener()或者detachEvent()方法来解除绑定。但是如果你的应用稍微复杂了点，绑定和需要解绑的事件处理程序一多，管理就成了痛苦的事情，你不仅需要知道那些函数绑定在了哪些元素上，还得一个一个来解除绑定。那么这时候用这个方法当然就不能忍了，我们就需要一种能够方便管理绑定在元素身上的事件处理函数的方法。

基本思路

我们知道事件处理程序是绑定在元素身上的，但是绑定了之后我们居然没有办法查看他身上绑定了哪些甚至多少事件处理函数！

所以问题把我们导向了存储绑定信息上。我们可以在把函数绑定在了元素上之后，将这个过程记录下来。因为每个事件处理函数都会对应一个元素，因此我们要通过元素访问事件处理程序的话就需要将事件处理程序记录在元素的某个属性中。我们可以通过给每个要绑定事件处理程序的元素创建一个记录信息的属性来实现，为了便于管理，我们可以创建一个公共对象，存放各个元素的信息，每个元素只用提供一个唯一的id（guid）就可以与之对应。然后我们就可以在绑定事件处理程序之后将事件处理程序的引用保存在公共信息管理对象中对应元素guid的那个子对象里面，这样就可以在之后对元素绑定过的事件处理程序进行操作和访问了。下面先来看看记录函数信息的代码：

var KM = function(){
    //＊＊＊＊＊＊＊＊纪录DOM元素相关信息＊＊＊＊＊＊＊＊
    var cache = {},                            //私有对象属性，存放元素的guid，每个guid对应的值为一个存放元素信息的对象
            guidCounter = 1,                //私有变量，为每个元素分配guid
            expando = "data-" + (new Date()).getTime();   //私有变量，为每个元素的自定义特性分配名字（根据创建的时间）

    inKM.getData = function(element) {        //KM.getData(elem)，公有方法，读取传入元素的信息
        var guid = element[expando];                //设置guid为元素对应expando特性的值（如果存在的话）
        if(!guid){                                                    //如果guid不存在，即元素中没有expando特性
            guid = element[expando] = guidCounter++;
            cache[guid] = {};                                    //初始化缓存中对应元素guid的信息对象
        }
        return cache[guid];
    };

    inKM.removeData = function(element) {   //KM.removeData(elem)，公有方法，清除元素的绑定信息和存放guid的expando特性
        var guid = element[expando];
        if(!guid) {                                                        //传入的元素本来就没绑定过信息，没有guid
            return;
        }
        delete cache[guid];
        try {
            delete element[expando];
        }
        catch (e) {
            if(element.removeAttribute) {
                element.removeAttribute(expando);
            }
        }
    };
    //＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊

    return inKM;
}();

上面代码是从我的工具函数中摘下来的，因此包装在了一个命名空间中，之后的代码也会是这个写法（太懒不想改了）。记录元素信息的代码实现中，在闭包中创建了公有方法getData()和removeData()用来获取和删除公共信息管理对象中对应这个元素的子对象。等会会看到我们就是把事件处理函数装在这个子对象中的。私有变量cache就是公共信息管理对象。有了记录元素信息的东西，下面就可以来看看绑定事件处理程序和解绑的代码了：

var KM = function(){
    //＊＊＊＊＊＊＊＊绑定事件处理程序＊＊＊＊＊＊＊＊
    var nextGuid = 1;    //私有变量，为事件处理程序提供guid

    inKM.EventUtil.addEventListener = function(element, type, handler) {  //KM.EventUtil.addEventListener(elem,type,fn)，公有方法
        var data = KM.getData(element);                            //获取绑定在element身上的信息

        if(!data.dispatcher) {                                            //判断data中有没有调度函数，没有则创建
            data.disabled = false;                                        //禁用调度函数
            data.dispatcher = function(event) {
                if(data.disabled) return;
                event = KM.EventUtil.fixEvent(event);
                var handlers = data.handlers[event.type];
                if(handlers) {
                    for(var n = 0; n < handlers.length; n++) {    //遍历执行所有data中对应type的事件处理程序
                        handlers[n].call(element, event);                    //并且指定执行上下文为element
                    }
                }
            };
        }

        if(!data.handlers) {
            data.handlers = {};                                                            //创建handlers对象，存放所有对应type的事件处理函数组成的数组
        }

        if(!data.handlers[type]) {
            data.handlers[type] = [];                                                //创建该type的事件处理程序数组
            if(document.addEventListener) {                                    //将调度函数添加为事件处理程序
                element.addEventListener(type, data.dispatcher, false);
            }
            else if(document.attachEvent) {
                element.attachEvent("on" + type, data.dispatcher);
            }
        }

        if(!handler.guid) {
            handler.guid = nextGuid++;                                            //给实际事件处理函数绑定guid
        }

        data.handlers[type].push(handler);                                //将事件处理程序放进data中，dispatcher可以利用闭包访问最新的事件处理程序
    };
    //＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊

    //＊＊＊＊＊＊＊＊解绑事件处理程序＊＊＊＊＊＊＊＊
    inKM.EventUtil.removeEventListener = function(element, type, handler) {        //根据传入的参数个数决定解绑方式
        var data = KM.getData(element);

        if(!data.handlers) {        //传入的元素上没有绑定handlers信息，即没绑定过事件处理函数
            return;
        }

        var removeType = function(t) {        //内部函数，用于解绑元素上对应type的所有事件处理程序
            data.handlers[t] = [];                    //将对应type的事件处理函数的数组设为空数组
            KM.EventUtil.clearResource(element, t);    //进行资源清理
        };

        if(!type) {                                                //只传入了element参数
            for(var t in data.handlers) {        //遍历handlers对象将所有值设为空数组然后进行资源清理
                removeType(t);
            }
            return;
        }

        var handlers = data.handlers[type];
        if(!handlers) {                                        //传入的元素里对应type没有绑定事件处理函数
            return;
        }

        if(!handler) {                                        //只传入了前两个参数
            removeType(type);
        }

        if(handler.guid) {                                //传入的想要进行解绑的事件处理函数必须在之前设定过guid，即被绑定了之后才能解绑
            for(var i = 0; i < handlers.length; i++) {
                if(handlers[i].guid === handler.guid) {
                    handlers.splice(i--, 1);
                }
            }
        }
        KM.EventUtil.clearResource(element, type);
    };
    //＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊

    return inKM;
}();

在绑定事件处理程序的过程中，我们给传入的事件处理程序加了一个代表身份的guid属性，该属性用来跟踪事件处理程序，会在之后的解绑过程中用上。在绑定过程中，我们先读取绑定在元素上的信息，如果是第一次绑定，那么就创建一个handlers对象做为元素信息对象的属性，用来存放对应每个type的事件处理函数的数组。同理如果该type是第一次绑定事件处理程序，那么也要手动创建handlers[type]数组，用来存放该type的事件处理程序。
同时可以看到，我们给元素信息对象添加了一个dispatcher（调度）方法，在这个方法中，调用了所有handlers[type]中的事件处理程序。如果该type是第一次绑定事件处理程序，那么就将调度函数做为该元素该type的事件处理函数。也就是说，真正的事件处理函数根本就没有绑定到元素上，而是通过调度函数来调用各个事件处理函数的。这样做的好处可以在解绑过程中体现出来，如果我们想解除某个事件处理函数的绑定，我们只需要找到handlers[type]中对应guid的函数将他删除就好。因为调度函数存在于闭包中，data.handlers[type]的更新会实时反映到调度函数中。

再来看看解绑过程，该函数支持三种用法。如果只传入element，那么就清除该元素的所有事件处理程序，这个过程很好实现，只需要将该元素的信息对象中的handlers清空，再做一些判断清理资源工作就好了（清理资源由clearResource函数实现，等会给出代码）。第二种是如果传入元素和事件类型不传入事件处理程序，那么就将handlers[type]数组清空就行了。如果三个参数都传入了，就清除handlers[type]中对应与传入函数的guid属性相同的那个函数，解绑过程就完成了。可以看出调度函数给我们带来的便利。下面是清理资源函数：

var KM = function() {
    inKM.EventUtil.clearResource = function(element, type) {    //解除事件处理程序绑定之后用于清除空的对象数组等资源
        function isEmpty(obj) {        //内部函数用来判断一个对象是否为空
            for(var prop in obj) {
                return false;
            }
            return true;
        }

        var data = KM.getData(element);

        if(data.handlers[type].length === 0) {        //判断handlers中对应type的事件处理程序数组是否为空，为空则清除数组和解除绑定此type下的调度函数
            delete data.handlers[type];
            if(document.removeEventListener) {
                element.removeEventListener(type, data.dispatcher, false);
            }
            else if(document.detachEvent) {
                element.detachEvent("on" + type, data.dispatcher);
            }
        }

        if(isEmpty(data.handlers)) {            //判断handlers对象是否为空，为空则说明该元素上没有绑定任何事件处理函数
            delete data.handlers;
            delete data.dispatcher;
        }

        if(isEmpty(data)) {                //判断data是否为空，为空说明此元素没有绑定任何数据
            KM.removeData(element);
        }
    }
    return inKM;
}();

在清理函数中，我们判断了handlers[type]数组中还有没有事件处理函数，如果没有了就删除这个数组，并且将调度函数从元素上解除绑定。然后判断handlers对象是否为空对象，如果为空则说明该元素上已经没有了事件处理程序，可以将信息对象上的handlers对象和dispatcher函数删除掉。最后判断元素的信息对象是否为空对象，如果是空对象就可以调用之前的KM.removeData()将信息对象从元素身上清除，达到清除资源的目的。

总结

综上所述，我们可以通过公共事件处理程序管理对象来管理繁琐的DOM事件模型。这样不仅给我们的代码编写带来了更多的便利，而且还更优雅。

自定义事件解耦

不知不觉写了这么多，这俩主题写一堆原来这么多，分开又不太合适，继续写吧

说到自定义事件就要说到解耦，说到解耦就离不开松耦合(loose coupling)和紧耦合(close coupling)。那就从这俩概念说起。

松耦合和紧耦合

首先让我先假设一个应用情景，我们需要在页面中执行一个ajax操作，并且在ajax刚开始的时候显示一个图片，ajax结束的时候将图片隐藏（应该是个相当常见的情景了）。好那么问题来了，用什么来判断并且执行显示和隐藏的操作呢？
草鸟们说，这也算是问题？在ajax刚开始的时候操作图片让他显示，在ajax回调中操作图片让他隐藏不就完了，太简单了。

<!DOCTYPE html>
<html>
      <head>
        <title>菜鸟的代码</title>
        <meta charset="utf-8">
        <style type="text/css">
              #whirlyThing { display: none; }
        </style>
        <script type="text/javascript">
              KM.EventUtil.addEventListener(window,'load',function(){
                var button = document.getElementById('clickMe');
                KM.EventUtil.addEventListener(button,'click',function(){
                      performAjaxOperation(this);
                });
            });
              function performAjaxOperation(target) {
                  document.getElementById('whirlyThing').style.display = 'inline-block'; //ajax操作刚开始
                  setTimeout(function(){            //定时器模拟ajax操作
                      document.getElementById('whirlyThing').style.display = 'none';    // ajax操作执行完毕，回调函数中隐藏图片
                  }, 5000);
              }
        </script>
      </head>
      <body>
        <button type="button" id="clickMe">Start</button>

        <img id="whirlyThing" src="whirly-thing.gif">
      </body>
</html>

这样是能实现我们的要求没错，但是如果这样处理的话，毫无代码重用性可言（都还没涉及到耦不耦合）。你好歹也把显示和隐藏操作代码封装一下吧，万一要用于其他地方你就调用一个函数就行了，不然你必须得复制整段代码啊，复制代码是禁忌啊，啊啊啊。所以，如果想让显示和隐藏图片的操作代码可以重用，至少应该这样写：

<!DOCTYPE html>
<html>
    <head>
        <title>菜鸟的改进</title>
        <meta charset="utf-8">
        <style type="text/css">
              #whirlyThing { display: none; }
        </style>

        <script type="text/javascript">

            KM.EventUtil.addEventListener(window,'load',function(){

                var button = document.getElementById('clickMe');
                KM.EventUtil.addEventListener(button,'click',function(){
                  performAjaxOperation(this);
                });
              });

              function performAjaxOperation(target) {
                  showImg(); //ajax操作刚开始

                  setTimeout(function(){            //定时器模拟ajax操作
                      hideImg();    // ajax操作执行完毕，回调函数中隐藏图片
                  }, 5000);
              }

              function showImg() {
                  document.getElementById('whirlyThing').style.display = 'inline-block';
              }
              function hideImg() {
                  document.getElementById('whirlyThing').style.display = 'none';
              }

        </script>
      </head>
      <body>

        <button type="button" id="clickMe">Start</button>

        <img id="whirlyThing" src="whirly-thing.gif">

      </body>
</html>

这样的代码看起来要比之前的简洁一些，showImg()和hideImg()两个函数也可以进行重用。那么现在就要说代码耦和的问题了。比如说这个ajax操作很复杂，又用得很广泛，于是我们就需要将这个代码放进一个单独的ajaxPerform.js文件中，然后在每个要用到这个ajax操作的页面里引用该文件。在这样的情景下，假设我想在自己的页面中用这个ajax操作，于是我就要引用ajaxPerform.js文件。为了能够实现该文件中的功能，我居然要在我的页面中重新定义两个相同名字的showImg()和hideImg()函数，才能实现相应的操作。两个还好，操作多了呢？每个页面都要定义跟ajaxPerform.js文件中一样名字的函数，就会相当的费力且麻烦。这就是代码的紧耦合，体现在在判断触发条件的代码中（也就是ajaxPerform.js文件的代码中）必须要知道触发事件之后的操作代码（也就是showImg()和hideImg()函数）的细节。这样ajaxPerform.js文件的代码不仅要负责判断什么时候触发条件，还要知道触发条件的时候要调用哪些函数。
举个例子，代码紧耦合中的performAjaxOperation()函数就像高中老师，上课了要叫班上每一位同学赶紧屁股坐端眼睛瞪大老子要上课了你们必须听，每个人他都管。班上所有同学都是上课事件触发时他要调用的函数。请问您累不累？

而代码松耦合中的performAjaxOperation()函数就像大学老师，上课了就开始讲，你爱听不听，我一个都不管，我只管讲我的课。感兴趣的同学（元素）就注册老师上课事件，老师上课的时候就调用事件处理函数（滚去上课）。不感兴趣的同学（元素）不会注册老师上课事件，老师上课事件触发的时候自然就没有他们什么事了。

没错，正如上面说的代码松耦合，要实现松耦合就要用到事件。显然浏览器没有提供老师上课这个默认事件，因此老师上课这个事件就是个自定义事件。既然浏览器没有这个事件，也就不会自动触发了，所以我们需要编写事件触发的代码：

var KM = function(){
    inKM.EventUtil.triggerEvent ＝ function(element, event) {                                //在指定元素上触发冒泡事件
    //传入的event可以是event对象，也可以是字符串（比如说自定义事件）
        var elementData = KM.getData(element),                                //获取绑定在元素上的信息（主要用来获取dispatcher调度函数）
              parent = element.parentNode || element.ownerDocument;    //得到元素的父节点(最后得到顶层的document)

        if(typeof event === "string") {                                                //如果传入的字符串就生成一个自定义event对象
            event = {type: event, target: element};
        }
        event = KM.EventUtil.fixEvent(event);

        if(elementData.dispatcher) {                                                    //调用调度函数来执行对应event的事件处理程序
            elementData.dispatcher.call(element, event);
        }

        if (parent && !event.isPropagationStopped()) {        //确保父节点存在并且没有阻止事件冒泡，接着触发父节点上的事件
            KM.EventUtil.triggerEvent(parent, event);
        }

          else if (!parent && !event.isDefaultPrevented()) {    //到了最顶层的document，确保浏览器默认行为没有被禁用，然后执行默认行为
            var targetData = KM.getData(event.target);

            if (event.target[event.type]) {                     //如果元素在该事件上有默认行为
                targetData.disabled = true;                       //在执行默认行为之前禁用调度函数，因为会触发该事件

                event.target[event.type]();                       //执行默认行为

                targetData.disabled = false;                      //重新将调度函数设为可用
            }
        }
    }
}();

触发自定义事件

有了事件触发函数，我们就可以在选定元素上触发一个冒泡事件了。输入的event可以是字符串，也可以是标准event对象。在一个元素上触发一个事件会从该元素到元素所在的document对象一路冒泡上去，经过的每一个父节点都会对该事件作出响应。有了事件触发，我们就可以触发我们的自定义事件了。于是回到之前的ajax话题，最后通过自定义事件解耦之后的代码是这样的：

<!DOCTYPE html>
<html>
    <head>
        <title>自定义事件解耦</title>
        <meta charset="utf-8">
        <style type="text/css">
              #whirlyThing { display: none; }
        </style>

        <script type="text/javascript">

            KM.EventUtil.addEventListener(window,'load',function(){

                var button = document.getElementById('clickMe');
                KM.EventUtil.addEventListener(button,'click',function(){
                      performAjaxOperation(this);
                });

                var body = document.getElementsByTagName('body')[0];

                KM.EventUtil.addEventListener(body,'ajax-start',function(e){
                      document.getElementById('whirlyThing').style.display = 'inline-block';
                });

                KM.EventUtil.addEventListener(body,'ajax-complete',function(e){
                      document.getElementById('whirlyThing').style.display = 'none';
                });
              });

              function performAjaxOperation(target) {
                  KM.EventUtil.triggerEvent(target, "ajax-start"); //ajax操作刚开始

                  setTimeout(function(){            //定时器模拟ajax操作
                      KM.EventUtil.triggerEvent(target, "ajax-complete");    // ajax操作执行完毕，回调函数中隐藏图片
                  }, 5000);
              }

        </script>
      </head>
      <body>

        <button type="button" id="clickMe">Start</button>

        <img id="whirlyThing" src="whirly-thing.gif">

      </body>
</html>

这样，performAjaxOperation()函数就只管在ajax开始的时候分发事件，不管也不知道有没有人注册，一副爱谁谁的洒脱模样。这样我们就能在不同的页面中重用ajaxPerform.js文件而不用在自己页面中申明一大堆函数了，只需要将需要在事件触发时进行的操作放在事件处理函数中并且注册文件中的ajax事件就行了。

总结

说了这么多，也不知道说清楚没有，也不知道有没有理解有误的地方。反正自定义事件是个缓解代码紧耦合的好东西就是了。耦合一松，全身轻松。

describe("KM.getData(element) method", function(){

                }

        expect(data1.dispatcher instanceof Function).toBe(true);
    });

        var domElem = document.createElement("div"),

describe("KM.EventUtil.removeEventListener(element, type, handler) method", function(){

    it("can remove the handler from the array, which is KM.getData(element).handlers[type]", function(){
        var domElem = document.createElement("div"),
                type = "click",
                data = KM.getData(domElem),
                listener1 = function(){},
                listener2 = function(){};

        KM.EventUtil.addEventListener(domElem, type, listener1);
        KM.EventUtil.addEventListener(domElem, type, listener2);

        expect(data.handlers[type].length).toBe(2);
        KM.EventUtil.removeEventListener(domElem, type, listener1);

describe("KM.EventUtil.triggerEvent(element, event) method", function(){
    var domElem = document.createElement("div"),
            parentElem = document.createElement("div"),
            type = "speak",
            data = KM.getData(domElem),
        listener1,
        listener2;

    parentElem.appendChild(domElem);
    listener1 = jasmine.createSpy('listener1');
    listener2 = jasmine.createSpy('listener2');

    it("can trigger the specific event on the element", function(){
        KM.EventUtil.addEventListener(domElem, type, listener1);
        expect(listener1).not.toHaveBeenCalled();

        KM.EventUtil.triggerEvent(domElem, type);
        expect(listener1).toHaveBeenCalled();
    });

    it("and the event can propagate", function(){
        KM.EventUtil.addEventListener(parentElem, type, listener2);
        expect(listener2).not.toHaveBeenCalled();

        KM.EventUtil.triggerEvent(domElem, type);
        expect(listener2).toHaveBeenCalled();
    });
